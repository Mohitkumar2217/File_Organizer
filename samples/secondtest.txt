SELECT SUPPLIER_NAME
   FROM SUPPLIER
   RIGHT JOIN CATALOGSS
   ON SUPPLIER.SUPPLIER_ID = CATALOGSS.SUPPLIER_ID
   RIGHT JOIN PARTS
   ON CATALOGSS.PART_ID = PARTS.PART_ID
   WHERE PARTS.PART_COLOR = 'BLACK';
SELECT SUPPLIER_NAME
   FROM SUPPLIER
   RIGHT JOIN CATALOGSS
   ON SUPPLIER.SUPPLIER_ID = CATALOGSS.SUPPLIER_ID
   RIGHT JOIN PARTS
   ON CATALOGSS.PART_ID = PARTS.PART_ID
   WHERE PARTS.PART_COLOR = 'BLACK'AND PARTS.PART_COLOR = 'GREEN';
